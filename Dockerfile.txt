# Use a Node.js image
FROM node:18

# Install FFmpeg (required for music)
RUN apt-get update && apt-get install -y ffmpeg

# Create working directory
WORKDIR /app

# Copy package files first
COPY package*.json ./

# Install dependencies
RUN npm install

# Copy the rest of your bot files
COPY . .

# Start the bot
CMD ["node", "index.js"]
